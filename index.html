如何调用OpenAI API开发AI应用？

以下是完整的OpenAI API调用示例（Python）：


# OpenAI API国内直连调用示例
import requests

API_KEY = "your_api_key_here"  # 替换成您的API密钥
API_URL = "https://api.zzzzapi.com/v1/chat/completions"  # OpenAI API中转地址

headers = {
    "Authorization": f"Bearer {API_KEY}",
    "Content-Type": "application/json"
}

# 构建请求参数 - 与官方OpenAI API格式完全一致
data = {
    "model": "gpt-4o",  # 最新GPT-4o模型，也支持gpt-4-turbo、gpt-3.5-turbo等
    "messages": [
        {"role": "system", "content": "你是一个专业的AI助手，擅长解答OpenAI API相关问题"},
        {"role": "user", "content": "请详细介绍API中转站的优势和使用方法"}
    ],
    "temperature": 0.7,
    "max_tokens": 2000
}

# 发送请求到OpenAI API中转服务器
response = requests.post(API_URL, headers=headers, json=data)

# 解析返回结果
if response.status_code == 200:
    result = response.json()
    ai_response = result["choices"][0]["message"]["content"]
    print(ai_response)
else:
    print(f"请求失败: {response.status_code}")
    print(response.text)
